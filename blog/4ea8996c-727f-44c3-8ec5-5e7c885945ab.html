                </div>
            </div>
        </div>
    </div>
</template>

<script>
    import Vue from "vue";
    import { mapActions } from "vuex";
    
    export default {
        name: "Alert",
        props: {
            title: String,
            okButtonText: String,
        },
        methods: {
            ...mapActions({
                setInvisible: "SET_ALERT_INVISIBLE",
                setVisible: "SET_ALERT_VISIBLE"
            }),
            onOk() {
                this.$emit("ok");
                this.setInvisible();
            }
        },
        mounted: {
            Vue.prototype.$alert = (title, onOk = () => null) => this.setVisible({ show: true, title, okButtonText: "확인", onOk });
        }
    }
</script>

Pagination (Props로만 쓸 수 있는 것)

<!-- Pagination.vue -->
<template>
    <div id="pagination">
        <a class="prev" @click="onPrev" v-if="part > 1" v-html="prevButton"></a>
        <a :class="['page', _page === page ? 'on' : '']" :key="i" v-for="(_page, i) in pages" @click="onPage(_page)">{{ _page + 1 }}</a>
        <a class="next" @click="onNext" v-if="part < totalPartCount" v-html="nextButton"></a>
    </div>
</template>

<script>
    export default {
        name: "Pagination",
        data: () => ({
            page: 0,
            part: 1
        }),
        props: {
            prevButton: String,
            nextButton: String,
            totalCount: Number,
            size: Number,
            pageCount: Number
        },
        computed: {
            totalPageCount() {
                return parseInt(this.totalCount / this.size) + (this.totalCount % this.size !== 0);
            },
            totalPartCount() {
                return parseInt(this.totalPageCount / this.pageCount) + (this.totalPageCount % this.pageCount !== 0);
            },
            pages() {
                let pages = [];
                for (let i = (this.part - 1) * this.pageCount; i < this.part * this.pageCount && i < this.totalPageCount; ++i)
                    pages.push(i);
                return pages;
            }
        },
        watch: {
            size() {
                this.page = 0;
            }
        },
        methods: {
            onPrev() {
                this.page = --this.part * this.pageCount - 1;
                this.$emit("prev", this.page, this.size);
            },
            onPage(page) {
                this.page = page;
                this.$emit("page", this.page, this.size);
            },
            onNext() {
                this.page = (++this.part - 1) * this.pageCount;
                this.$emit("next", this.page, this.size);
            }
        }
    }
</script>
