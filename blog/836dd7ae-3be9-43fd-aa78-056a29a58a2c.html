Refactoring

회사에서든 혼자든 개발/유지보수를 하다보면 리팩토링(Refactoring)이라는 과정을 꼭 하곤 한다.

리팩토링(Refactoring)이란, 코드의 내부 동작을 효율적이고 깔끔하게 하여 유지보수의 생산성을 올리되, 코드 동작의 결과는 이전과 똑같이 하는 과정을 의미한다. 그 중 하나는 이름 바꾸기라던가, 불필요한 코드를 없앤다던가 등이 있는데, 프로젝트의 규모가 방대할 수록 사람의 손으로 일일히 다 찾아서 수정하는 것은 불가능에 가깝다.

이 페이지에서는 원하는 부분의 코드 수정을 단시간에 자동으로 해주는 툴을 소개하려고 한다.

JSCodeShift

아래는 하나의 예시다. 실제로 내가 필요에 의해 진행했던 작업이다.

JSX 엘리먼트 중 color="blue600" 이라는 Attribute를 갖는 엘리먼트를 찾아서 color={Colors.blue['600']} 로 바꾸고, 이 때 Colors 라는 변수가 import 되어야하기 때문에 만약 import 되어있지 않으면 import 해주는 코드를 최상단에 추가하도록 하고 싶었다.

// transform.js

/** @type import('jscodeshift').Transform */
function transform(file, { j }) {
	// 파일의 내용 파싱
  const tree = j(file.source);

	// AS-IS
  const src = 'blue600';
  // TO-BE
  const dst = "{Colors.blue['600']}";

  const nodes = tree.find(
	  // JSX Attribute가
	  j.JSXAttribute,
	  // `color="blue600"` 인 엘리먼트를 찾는다.
	  node => {
	    return node.name.name === 'color' && node.value.value === src;
	  },
  );

	// 찾음
  if (nodes.length > 0) {
	  // `'@/constants/color'` 로부터 import 하는 코드 존재 여부
	  // Colors 변수가 이 source에 있음
    const colorsImported = tree.find(
	    j.ImportDeclaration,
	    node => {
		    return node.source.value === '@/constants/color';
	    },
	  );
	  /**
	   * 만약 `'@/constants/color'` 로부터 import 되어있지 않으면
	   * 파일 내 최상단에 import statement 추가
	   */
    if (colorsImported.length === 0) {
	    // 파일 내 모든 import 문 중
      tree.find(j.ImportDeclaration)
	      // 첫 번째 라인
	      .at(0)
	      // 윗 줄에 추가
	      .insertBefore(
		      j.importDeclaration(
			      // `Colors` 변수를 default로 import
			      [j.importDefaultSpecifier(j.identifier('Colors'))],
			      // `'@/constants/color'` 로부터
			      j.literal('@/constants/color'),
			    ),
			  );
    }

		// `color={Colors.blue['600']}` 라는 코드로 치환
    nodes.forEach(node => {
      node.replace(j.jsxAttribute(j.jsxIdentifier('color'), j.jsxText(dst)));
    });
  }

	// 코드 내보내기
  return tree.toSource();
}
